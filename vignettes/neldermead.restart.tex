\HeaderA{neldermead.restart}{Restart neldermead search.}{neldermead.restart}
\keyword{method}{neldermead.restart}
%
\begin{Description}\relax
Update the simplex with \code{neldermead.updatesimp} and restart the search
with \code{neldermead.search}.
\end{Description}
%
\begin{Usage}
\begin{verbatim}
  neldermead.restart(this = NULL)
\end{verbatim}
\end{Usage}
%
\begin{Arguments}
\begin{ldescription}
\item[\code{this}] A neldermead object.
\end{ldescription}
\end{Arguments}
%
\begin{Value}
Returns an updated neldermead object.
\end{Value}
%
