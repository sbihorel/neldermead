\HeaderA{neldermead.get}{Get the value for the given element}{neldermead.get}
\keyword{method}{neldermead.get}
%
\begin{Description}\relax
Get the value for the given element in a neldermead object.
\end{Description}
%
\begin{Usage}
\begin{verbatim}
  neldermead.get(this = NULL, key = NULL)
\end{verbatim}
\end{Usage}
%
\begin{Arguments}
\begin{ldescription}
\item[\code{this}] A neldermead object.
\item[\code{key}] The name of the key to query.
\end{ldescription}
\end{Arguments}
%
\begin{Value}
Return the value of the list element \code{key}, or an error message if
\code{key} does not exist in the neldermead object \code{this}.
\end{Value}
%
