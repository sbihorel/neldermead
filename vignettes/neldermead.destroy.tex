\HeaderA{neldermead.destroy}{Erase a neldermead object.}{neldermead.destroy}
\keyword{method}{neldermead.destroy}
%
\begin{Description}\relax
\code{neldermead.destroy} calls \code{optimbase.destroy} and
\code{optimsimplex.destroy} to erase the content of \code{this\$optbase} and
\code{this\$simplex0}.
\end{Description}
%
\begin{Usage}
\begin{verbatim}
  neldermead.destroy(this = NULL)
\end{verbatim}
\end{Usage}
%
\begin{Arguments}
\begin{ldescription}
\item[\code{this}] A neldermead object.
\end{ldescription}
\end{Arguments}
%
\begin{Value}
Return an updated neldermead object.
\end{Value}
%
